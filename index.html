Google Sheets Setup

To enable form submissions to Google Sheets:

1. Open Google Sheets and go to Extensions → Apps Script
2. Paste the Apps Script code (copy below)
3. Deploy as Web App (access: Anyone)
4. Copy the Web App URL and paste it here

function doPost(e) { var sheet = SpreadsheetApp.openById('YOUR_SHEET_ID'); var data = JSON.parse(e.postData.contents); var tab = sheet.getSheetByName(data.sheet) || sheet.insertSheet(data.sheet); if (tab.getLastRow() === 0) tab.appendRow(data.headers); tab.appendRow(data.row); return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput('IVO API active'); }
